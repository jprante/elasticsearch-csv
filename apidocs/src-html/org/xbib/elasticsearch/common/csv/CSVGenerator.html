001package org.xbib.elasticsearch.common.csv;
002
003import java.io.Closeable;
004import java.io.Flushable;
005import java.io.IOException;
006import java.io.Writer;
007import java.util.List;
008
009public class CSVGenerator implements Closeable, Flushable {
010
011    private final static char comma = ',';
012
013    private final static char quote = '\"';
014
015    private final static char escape = '\"';
016
017    private final static char tab = '\t';
018
019    private final static String lf = System.getProperty("line.separator");
020
021    private Writer writer;
022
023    private int col;
024
025    private String[] keys;
026
027    public CSVGenerator(Writer writer) {
028        this.writer = writer;
029        this.col = 0;
030        this.keys = new String[]{};
031    }
032
033    public CSVGenerator keys(List<String> keys) {
034        this.keys = keys.toArray(new String[keys.size()]);
035        return this;
036    }
037
038    public CSVGenerator writeKeys() throws IOException {
039        for (String k : keys) {
040            write(k);
041        }
042        return this;
043    }
044
045    public CSVGenerator write(String value) throws IOException {
046        if (col > 0) {
047            writer.write(comma);
048        }
049        writer.write(escape(value));
050        col++;
051        if (col >= keys.length) {
052            nextRow();
053        }
054        return this;
055    }
056
057    public CSVGenerator nextRow() throws IOException {
058        if (col > 0) {
059            writer.write(lf);
060            col = 0;
061        }
062        return this;
063    }
064
065    @Override
066    public void close() throws IOException {
067        writer.close();
068    }
069
070    @Override
071    public void flush() throws IOException {
072        writer.flush();
073    }
074
075    private String escape(String value) {
076        if (value == null) {
077            return "";
078        }
079        if (value.indexOf(quote) < 0 && value.indexOf(escape) < 0
080                && value.indexOf(comma) < 0 && value.indexOf(tab) < 0  && !value.contains(lf)) {
081           return value;
082        }
083        int length = value.length();
084        StringBuilder sb = new StringBuilder(length + 2);
085        sb.append(quote);
086        for (int i = 0; i < length; i++) {
087            char ch = value.charAt(i);
088            if (ch == quote) {
089                sb.append(quote);
090            }
091            sb.append(ch);
092        }
093        sb.append(quote);
094        return sb.toString();
095    }
096
097}
