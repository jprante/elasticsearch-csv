001package org.xbib.elasticsearch.rest.csv;
002
003import org.elasticsearch.action.search.SearchRequest;
004import org.elasticsearch.client.Client;
005import org.elasticsearch.common.collect.Sets;
006import org.elasticsearch.common.inject.Inject;
007import org.elasticsearch.common.settings.Settings;
008import org.elasticsearch.rest.BaseRestHandler;
009import org.elasticsearch.rest.RestChannel;
010import org.elasticsearch.rest.RestController;
011import org.elasticsearch.rest.RestRequest;
012import org.elasticsearch.rest.action.search.RestSearchAction;
013
014import java.util.Set;
015
016import static org.elasticsearch.rest.RestRequest.Method.GET;
017import static org.elasticsearch.rest.RestRequest.Method.POST;
018
019public class CSVRestSearchAction extends BaseRestHandler {
020
021    @Inject
022    public CSVRestSearchAction(Settings settings, Client client, RestController controller) {
023        super(settings, client);
024        controller.registerHandler(GET, "/_search_csv", this);
025        controller.registerHandler(POST, "/_search_csv", this);
026        controller.registerHandler(GET, "/{index}/_search_csv", this);
027        controller.registerHandler(POST, "/{index}/_search_csv", this);
028        controller.registerHandler(GET, "/{index}/{type}/_search_csv", this);
029        controller.registerHandler(POST, "/{index}/{type}/_search_csv", this);
030        controller.registerHandler(GET, "/_search_csv/template", this);
031        controller.registerHandler(POST, "/_search_csv/template", this);
032        controller.registerHandler(GET, "/{index}/_search_csv/template", this);
033        controller.registerHandler(POST, "/{index}/_search_csv/template", this);
034        controller.registerHandler(GET, "/{index}/{type}/_search_csv/template", this);
035        controller.registerHandler(POST, "/{index}/{type}/_search_csv/template", this);
036    }
037
038    @Override
039    public void handleRequest(final RestRequest request, final RestChannel channel) {
040        SearchRequest searchRequest = RestSearchAction.parseSearchRequest(request);
041        searchRequest.listenerThreaded(false);
042        String[] s = request.paramAsStringArray("keys", null);
043        Set<String> keys = s == null ? null : Sets.newHashSet(s);
044        client.search(searchRequest, new CSVToXContentListener(channel, keys));
045    }
046
047}
